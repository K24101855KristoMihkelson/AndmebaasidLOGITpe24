-- 1. Osa: andmebaasi ja tabelite koostamine 

-- Koostan uue andmebaasi.
CREATE DATABASE KristoHinne3;
USE KristoHinne3;

-- Koostan tabeli lennujaam, millel on lennujaamade andmed.
CREATE TABLE LennuJaam (
    LennuJaamID INT PRIMARY KEY IDENTITY(1,1),
    LennujaamaNimi VARCHAR(50),
    Linn VARCHAR(50)
);
-- Koostan tabeli lend, mis kuvab lennunumbri, väljumisaja ning ka lennujaama id.
CREATE TABLE Lend (
    LendID INT PRIMARY KEY IDENTITY(1,1),
    LennuNumber VARCHAR(20),
    Valjumisaeg DATETIME,
    LennujaamID INT,
    FOREIGN KEY (LennujaamID) REFERENCES Lennujaam(LennujaamID)
);
-- Koostan tabeli, mis näitab reisijaid, nime ja piletinumbrit.
CREATE TABLE Reisija (
    ReisijaID INT PRIMARY KEY IDENTITY(1,1),
    Nimi VARCHAR(50),
    Piletinumber VARCHAR(20),
    LendID INT,
    FOREIGN KEY (LendID) REFERENCES Lend(LendID)
);
-- Koostan logitabeli, mis hakkab automaatselt salvestama muudatusi.
CREATE TABLE logi (
    id INT PRIMARY KEY IDENTITY(1,1),
    kasutaja VARCHAR(50),
    kuupaev DATETIME,
    sisestusAndmed VARCHAR(255)
);
-- Koostan uue tabeli lennufirmade jaoks.
CREATE TABLE Lennufirma (
    LennufirmaID INT PRIMARY KEY IDENTITY(1,1),
    FirmaNimi VARCHAR(50),
    Luhikood VARCHAR(10) 
);

-- 3. Osa: trigerite koostamine (admini roll)

-- Koostan trigeri lennu lisamise jälgimiseks.
CREATE TRIGGER lennu_lisamine
ON Lend
AFTER INSERT
AS
BEGIN
    INSERT INTO logi (kasutaja, kuupaev, sisestusAndmed)
    SELECT CURRENT_USER, GETDATE(), LennuNumber 
    FROM inserted;
END;


-- Koostan trigeri lennu kustutamise jälgimiseks.
CREATE TRIGGER lennu_kustutamine
ON Lend
AFTER DELETE
AS
BEGIN
    INSERT INTO logi (kasutaja, kuupaev, sisestusAndmed)
    SELECT CURRENT_USER, GETDATE(), LennuNumber 
    FROM deleted;
END;

-- Koostan protseduuri, mis lisab uue lennujaama.
CREATE PROCEDURE LisaLennujaam
    @nimi VARCHAR(50),
    @linn VARCHAR(50)
AS
BEGIN
    INSERT INTO Lennujaam (LennujaamaNimi, Linn) VALUES (@nimi, @linn);
END;


-- Koostan protseduuri, mis otsib lendu linna järgi.
CREATE PROCEDURE OtsiLendu
    @otsitavLinn VARCHAR(50)
AS
BEGIN
    SELECT Lend.LennuNumber, Lend.Valjumisaeg
    FROM Lend
    JOIN Lennujaam ON Lend.LennujaamID = Lennujaam.LennujaamID
    WHERE Lennujaam.Linn = @otsitavLinn;
END;


-- Koostan protseduuri, mis kustutab reisija id järgi.
CREATE PROCEDURE KustutaReisija
    @reisijaID INT
AS
BEGIN
    DELETE FROM Reisija WHERE ReisijaID = @reisijaID;
END;


-- Koostan esimese vaate, mis näitab ainult tallinna lende.
CREATE VIEW vw_TallinnaLennud AS
SELECT Lend.LennuNumber, Lennujaam.LennujaamaNimi, Lennujaam.Linn
FROM Lend
JOIN Lennujaam ON Lend.LennujaamID = Lennujaam.LennujaamID
WHERE Lennujaam.Linn = 'Tallinn';


-- Koostan teise vaate, mis näitab konkreetse reisija info.
CREATE VIEW vw_PeetriPilet AS
SELECT Reisija.Nimi, Reisija.Piletinumber, Lend.LennuNumber
FROM Reisija
JOIN Lend ON Reisija.LendID = Lend.LendID
WHERE Reisija.Nimi = 'Peeter Puu';


-- Koostan kolmanda vaate, mis näitab ainult tulevikus toimuvaid reise.
CREATE VIEW vw_TulevasedReisid AS
SELECT Reisija.Nimi, Lend.LennuNumber, Lend.Valjumisaeg
FROM Reisija
JOIN Lend ON Reisija.LendID = Lend.LendID
WHERE Lend.Valjumisaeg > GETDATE();
